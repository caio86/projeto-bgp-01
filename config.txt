ip as-path access-list 30 permit ^30$
!
route-map TO-AS30 permit 10
set as-path prepend 20 20 20
match as-path 20
!
route-map FROM-AS30 permit 10
set local-preference 300
match as-path 30
!
route-map TO-AS100 permit 10
set as-path prepend 20 20 20
!

!
neighbor RR peer-group
neighbor RR remote-as 20
neighbor RR update-source lo0
neighbor RR next-hop-self
!
neighbor 1.1.1.1 peer-group RR
neighbor 4.4.4.4 peer-group RR
!
neighbor 6.6.6.6 remote-as 30
neighbor 6.6.6.6 ttl-security hops 1
neighbor 6.6.6.6 disable-connected-check
neighbor 6.6.6.6 update-source Loopback0
neighbor 6.6.6.6 route-map TO-AS30 out
neighbor 6.6.6.6 route-map FROM-AS30 in
!
neighbor 1.100.0.8 remote-as 100

!

network 2003:10:10::/64
neighbor FC00:100:10::8 activate
exit-address-family
!
end
